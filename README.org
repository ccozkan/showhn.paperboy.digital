** development
*** installation
    #+BEGIN_SRC
git clone git@github.com:ccozkan/showhn.paperboy.digital.git
cd showhn.paperboy.digital
rails db:create && rails db:migrate
foreman start
    #+END_SRC
*** running tests
    #+BEGIN_SRC
bundle exec rspec
    #+END_SRC
*** checking mails
    #+BEGIN_SRC
gem install mailcatcher
mailcatcher # check http://127.0.0.1:1080/ on a browser
